/- We define the reduced radial ODE system corresponding to the

/- The ODE system:
   α'' + (1/r) α' - (1/r²) α + ω² α = β
   β'' + (1/r) β' - (1/r²) β + ω² β = α - λ β - 4g β³
-/

def ode_system (p : ChaoitonParameters) (α β : ℝ → ℝ) : Prop :=
  ∀ (r : ℝ), r > 0 →
    (deriv (deriv α) r + (1/r) * deriv α r - (1/(r^2)) * α r + (p.ω)^2 * α r = β r) ∧
    (deriv (deriv β) r + (1/r) * deriv β r - (1/(r^2)) * β r + (p.ω)^2 * β r = α r - p.λ * β r - 4 * p.g * (β r)^3)

/- Boundary conditions:
   α(r) ~ A₀ r  as r → 0⁺
   β(r) ~ B₀ r  as r → 0⁺
   α(r) → 0    as r → ∞
   β(r) → 0    as r → ∞
-/

def boundary_conditions (p : ChaoitonParameters) (α β : ℝ → ℝ) : Prop :=
  (tendsto (λ r => α r / r) (𝓝[{0}ᶜ] (0 : ℝ)) (𝓝 p.A₀)) ∧
  (tendsto (λ r => β r / r) (𝓝[{0}ᶜ] (0 : ℝ)) (𝓝 p.B₀)) ∧
  (tendsto α atTop (𝓝 0)) ∧
  (tendsto β atTop (𝓝 0))
